        JunkCar j1 = new JunkCar("Charger","Dodge", 1987,"lightly damaged");
        vehicles[0] = a1;
        vehicles[1] = c1;
        vehicles[2] = s1;
        vehicles[3] = j1;

            System.out.println(i.toString());
        }

        for (Automobile i : vehicles) {
            i.honk();

        c1.accelerate();
        c1.stop();
        s1.accelerate();

        s1.supercharge();
